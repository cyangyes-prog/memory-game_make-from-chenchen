<script>
let emojis=["🍎","🍎","🐶","🐶","🚗","🚗","⚽","⚽"];
emojis.sort(()=>0.5-Math.random());

let first=null;
let second=null;

const grid=document.getElementById("grid");

emojis.forEach((emoji,index)=>{
    let div=document.createElement("div");
    div.className="card";
    div.dataset.value=emoji;
    div.innerText="❓";

    div.onclick=function(){
        if(div.innerText!="❓") return;

        div.innerText=emoji;

        if(!first){
            first=div;
        }else{
            second=div;

            if(first.dataset.value==second.dataset.value){
                first=null;
                second=null;
            }else{
                setTimeout(()=>{
                    first.innerText="❓";
                    second.innerText="❓";
                    first=null;
                    second=null;
                },800);
            }
        }
    }

    grid.appendChild(div);
});
</script>
